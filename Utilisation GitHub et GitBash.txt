$ git init	//Initialise le git

$ git add .	//Ajoute un repertoire

$ git commit -m "V 1"	//donne un numero de version

$ git remote add origin https://github.com/SteevenT/ExempleGit	//Mettre le lien du GitHub + demande de connexion

$ git push origin master	//Poussez la demande

$ ls		// Information du git

$ git push origin master -f	//-f pour forcer si une error

$ git checkout -b develop	//Créer une branche nommé develop et vas dessus

$ git branch develop		//Créer une branche nommé develop

$ git checkout develop		//Passe sur la branche develop

$ git checkout master		//retourne sur la branche principale

$ git merge develop -Steeven	//Ajout des modification de steeven dans la branche master

$ git merge develop -Laura	//Ajout des modification de Laura dans la branche master
